    <script>
        let currentStep = 1;
        const totalSteps = 5;
        const steps = [];
        for (let i = 1; i <= totalSteps; i++) {
            steps.push(document.getElementById(`step-${i}`));
        }
        const progressBar = document.getElementById('progressBar');
        const finalMessage = document.getElementById('finalMessage');
        const celebrateButton = document.getElementById('celebrateButton');
        const polaroidContainer = document.getElementById('polaroid-container');
        const polaroidElement = polaroidContainer ? polaroidContainer.querySelector('.polaroid') : null;

        // Initialize first step visibility
        gsap.set(steps.slice(1), { opacity: 0, scale: 0.95, y: 20 });
        gsap.set(steps[0], { opacity: 1, scale: 1, y: 0 }); // Ensure first step is active initially

        function updateProgressBar() {
            const progress = (currentStep - 1) / (totalSteps - 1) * 100;
            progressBar.style.width = `${progress}%`;
        }

        function nextStep() {
            if (currentStep < totalSteps) {
                gsap.to(steps[currentStep - 1], {
                    opacity: 0,
                    scale: 0.95,
                    y: -20,
                    duration: 0.6,
                    ease: "power2.out"
                });

                currentStep++;
                updateProgressBar();

                gsap.fromTo(steps[currentStep - 1],
                    { opacity: 0, scale: 0.95, y: 20 },
                    { opacity: 1, scale: 1, y: 0, duration: 0.6, ease: "power2.out", delay: 0.3 }
                );

                // Animate elements inside the new step with a stagger
                gsap.fromTo(steps[currentStep - 1].children,
                    { opacity: 0, y: 10 },
                    { opacity: 1, y: 0, duration: 0.5, ease: "power1.out", stagger: 0.1, delay: 0.6 }
                );
            }
        }

        function celebrate() {
            gsap.to(celebrateButton, { opacity: 0, duration: 0.3, onComplete: () => celebrateButton.disabled = true });
            gsap.to(finalMessage, { opacity: 1, y: 0, duration: 1.5, ease: "power2.out", delay: 0.5 });

            // Confetti bursts
            setTimeout(() => {
                confetti({
                    particleCount: 200,
                    spread: 120,
                    origin: { x: 0, y: 1 },
                    colors: ['#ff6b8f', '#ff4d79', '#ffe0f0', '#ffffff', '#8338ec', '#3a86ff']
                });
                confetti({
                    particleCount: 200,
                    spread: 120,
                    origin: { x: 1, y: 1 },
                    colors: ['#ff6b8f', '#ff4d79', '#ffe0f0', '#ffffff', '#8338ec', '#3a86ff']
                });
            }, 500);

            // Continuous confetti shower
            let confettiInterval = setInterval(() => {
                confetti({
                    particleCount: 50,
                    spread: 360,
                    startVelocity: 30,
                    scalar: 0.9,
                    gravity: 0.7,
                    ticks: 200,
                    origin: { x: Math.random(), y: Math.random() - 0.2 },
                    colors: ['#ff6b8f', '#ff4d79', '#ffe0f0', '#ffffff', '#8338ec', '#3a86ff'],
                    shapes: ['circle', 'square']
                });
                confetti({
                    particleCount: 20,
                    spread: 360,
                    startVelocity: 20,
                    scalar: 0.8,
                    gravity: 0.6,
                    ticks: 150,
                    origin: { x: Math.random(), y: Math.random() - 0.2 },
                    colors: ['#ff6b8f'], // Primarily pink for emoji
                    shapes: ['heart', 'star'] // Emoji confetti
                });
            }, 150);

            setTimeout(() => {
                clearInterval(confettiInterval);
                // Clear any remaining confetti after a delay if needed
                setTimeout(() => {
                    const canvas = document.querySelector('canvas.confetti-canvas');
                    if (canvas) canvas.remove();
                }, 3000);
            }, 5000); // Stop confetti after 5 seconds

            // Animate 3D hearts swirling outwards and fading
            heartMeshes.forEach((mesh, index) => {
                gsap.to(mesh.position, {
                    x: (Math.random() - 0.5) * 50, // Random X exit
                    y: 20 + Math.random() * 10, // Float up off screen
                    z: (Math.random() - 0.5) * 50, // Random Z exit
                    duration: 4,
                    ease: "power2.out",
                    delay: Math.random() * 0.5
                });
                gsap.to(mesh.material, {
                    opacity: 0,
                    duration: 3,
                    delay: Math.random() * 0.5 + 1,
                    onUpdate: function() {
                        mesh.material.transparent = true;
                    },
                    onComplete: () => {
                        scene.remove(mesh);
                        mesh.geometry.dispose();
                        mesh.material.dispose();
                    }
                });
            });
        }

        // Polaroid Parallax Effect
        if (polaroidElement) {
            polaroidContainer.addEventListener('mousemove', (e) => {
                const rect = polaroidContainer.getBoundingClientRect();
                const x = e.clientX - rect.left;
                const y = e.clientY - rect.top;

                const centerX = rect.width / 2;
                const centerY = rect.height / 2;

                const rotateY = (x - centerX) / centerX * 10; // Max 10 degrees rotation
                const rotateX = -((y - centerY) / centerY * 10); // Max 10 degrees rotation

                gsap.to(polaroidElement, {
                    rotationX: rotateX,
                    rotationY: rotateY,
                    duration: 0.1,
                    ease: "power1.out"
                });
            });

            polaroidContainer.addEventListener('mouseleave', () => {
                gsap.to(polaroidElement, {
                    rotationX: 0,
                    rotationY: 0,
                    duration: 0.3,
                    ease: "power1.out"
                });
            });
        }

        // --- Three.js for 3D Floating Hearts ---
        let scene, camera, renderer, heartMeshes = [];
        const numHearts = 25;
        const heartGeometry = new THREE.ShapeGeometry(
            new THREE.Shape()
                .moveTo(0.5, 0.2)
                .bezierCurveTo(0.5, 0.6, 0.1, 0.6, 0.1, 0.2)
                .bezierCurveTo(0.1, -0.2, 0.5, -0.2, 0.5, 0.2)
                .moveTo(-0.5, 0.2)
                .bezierCurveTo(-0.5, 0.6, -0.1, 0.6, -0.1, 0.2)
                .bezierCurveTo(-0.1, -0.2, -0.5, -0.2, -0.5, 0.2)
        );

        function initThreeJS() {
            scene = new THREE.Scene();
            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            camera.position.z = 10;

            renderer = new THREE.WebGLRenderer({
                canvas: document.getElementById('threejs-canvas'),
                alpha: true,
                antialias: true
            });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(window.devicePixelRatio);

            // Lights
            const ambientLight = new THREE.AmbientLight(0x404040, 2); // soft white light
            scene.add(ambientLight);
            const pointLight = new THREE.PointLight(0xffffff, 1);
            pointLight.position.set(5, 5, 5);
            scene.add(pointLight);

            // Create hearts
            for (let i = 0; i < numHearts; i++) {
                const material = new THREE.MeshPhongMaterial({
                    color: new THREE.Color(Math.random() * 0xff0000 + 0xdd00dd), // Pink/red shades
                    specular: 0x555555,
                    shininess: 30,
                    flatShading: false
                });
                const heart = new THREE.Mesh(heartGeometry, material);

                heart.position.x = (Math.random() - 0.5) * 20;
                heart.position.y = (Math.random() - 0.5) * 20;
                heart.position.z = (Math.random() - 0.5) * 20;

                heart.rotation.x = Math.random() * Math.PI;
                heart.rotation.y = Math.random() * Math.PI;
                heart.rotation.z = Math.random() * Math.PI;

                const scale = 0.5 + Math.random() * 0.8;
                heart.scale.set(scale, scale, scale);

                heartMeshes.push(heart);
                scene.add(heart);
            }

            animateThreeJS();
        }

        function animateThreeJS() {
            requestAnimationFrame(animateThreeJS);

            const time = Date.now() * 0.0005;

            heartMeshes.forEach(heart => {
                heart.position.y += Math.sin(time * 2 + heart.userData.randomFactor) * 0.005;
                heart.position.x += Math.cos(time * 1.5 + heart.userData.randomFactor) * 0.003;

                heart.rotation.x += 0.001;
                heart.rotation.y += 0.002;

                // Loop hearts if they go too far
                if (heart.position.y > 15) heart.position.y = -15;
                if (heart.position.y < -15) heart.position.y = 15;
                if (heart.position.x > 15) heart.position.x = -15;
                if (heart.position.x < -15) heart.position.x = 15;
            });

            renderer.render(scene, camera);
        }

        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        window.addEventListener('resize', onWindowResize, false);
        document.addEventListener('DOMContentLoaded', () => {
            initThreeJS();
            updateProgressBar(); // Initial progress bar state
        });
    </script>
